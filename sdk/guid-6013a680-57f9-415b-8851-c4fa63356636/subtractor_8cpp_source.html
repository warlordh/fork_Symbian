00157         {

00159         iMutex.Wait();

00168         iConsole->Printf(_L("Value read from the shared memory: %d\n"),*ptr);
00169 
00170         // Subtract the random value from the shared memory variable.
00171         *ptr -= randVal;

00173         while(*ptr < KLowerThreshold)
